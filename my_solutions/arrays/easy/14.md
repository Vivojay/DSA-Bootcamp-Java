# from typing import List

class Solution:
    def oddCells(self, m: int, n: int, indices: List[List[int]]) -> int:
        flattened_matrix = []
        count = 0
        matrix=self.define_matrix(m=m, n=n)

        for rc in indices:
            row, col = rc
            self.col_increment(matrix=matrix, col=col, m=m)
            self.row_increment(matrix=matrix, row=row, n=n)

        for row in matrix:
            flattened_matrix.extend(row)

        for num in flattened_matrix:
            # print(num)
            if num % 2 : # is odd
                count+=1
        return count

    # Create a null m x n matrix
    def define_matrix(self, m, n):
        matrix = []
        for i in range(m):
            res = [0 for i in range(n)]
            matrix.append(res)
        
        return matrix

    # Increment functions
    def col_increment(self, matrix, col, m):
        for i in range(m):
            matrix[i][col]+=1

    def row_increment(self, matrix, row, n):
        for i in range(n):
            matrix[row][i]+=1
